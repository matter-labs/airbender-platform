7pub fn generate(args: GenerateVkArgs) -> Result<()> {
8    ensure_gpu_vk_support()?;
9
10    let vk = match args.level {
11        ProverLevelArg::RecursionUnified => {
12            let vk = airbender_host::compute_unified_vk(&args.app_bin).map_err(|err| {
13                CliError::with_source(
14                    format!(
15                        "failed to compute unified verification keys for `{}`",
16                        args.app_bin.display()
17                    ),
18                    err,
19                )
20            })?;
21            airbender_host::VerificationKey::RealUnified(
22                airbender_host::RealUnifiedVerificationKey { vk },
23            )
24        }
25        ProverLevelArg::Base | ProverLevelArg::RecursionUnrolled => {
26            let level = as_host_level(args.level);
27            let vk = airbender_host::compute_unrolled_vk(&args.app_bin, level).map_err(|err| {
28                CliError::with_source(
29                    format!(
30                        "failed to compute unrolled verification keys for `{}`",
31                        args.app_bin.display()
32                    ),
33                    err,
34                )
35            })?;
36            airbender_host::VerificationKey::RealUnrolled(
37                airbender_host::RealUnrolledVerificationKey { level, vk },
38            )
39        }
40    };
41
42    write_bincode(&args.output, &vk)?;
43
44    ui::success("verification keys generated");
45    ui::field("level", level_name(args.level));
46    ui::field("output", args.output.display());
47
48    Ok(())
49}

51fn ensure_gpu_vk_support() -> Result<()> {
52    #[cfg(feature = "gpu-prover")]
53    {
54        Ok(())
55    }
56
57    #[cfg(not(feature = "gpu-prover"))]
58    {
59        Err(CliError::new(
60            "verification key generation requires GPU support in `cargo-airbender`",
61        )
62        .with_hint(
63            "rebuild `cargo-airbender` with default features or pass `--features gpu-prover` to use `generate-vk`",
64        ))
65    }
66}

68pub fn verify(args: VerifyProofArgs) -> Result<()> {
69    let expected_output_words = parse_expected_output_words(args.expected_output.as_deref())?;
70
71    let proof: airbender_host::Proof = read_bincode(&args.proof).map_err(|err| {
72        CliError::with_source(
73            format!("failed to decode proof from `{}`", args.proof.display()),
74            err,
75        )
76    })?;
77
78    let vk: airbender_host::VerificationKey = read_bincode(&args.vk).map_err(|err| {
79        CliError::with_source(
80            format!(
81                "failed to decode verification key file `{}`",
82                args.vk.display()
83            ),
84            err,
85        )
86    })?;
87
88    let level = match &proof {
89        airbender_host::Proof::Dev(_) => {
90            return Err(CliError::new(
91                "detected a dev proof; `cargo airbender verify-proof` supports only real proofs",
92            )
93            .with_hint(
94                "verify dev proofs through `airbender-host` with `Program::dev_verifier()`",
95            ));
96        }
97        airbender_host::Proof::Real(proof) => {
98            let expected_output_commit = expected_output_words
99                .as_ref()
100                .map(|words| words as &dyn airbender_host::Commit);
101
102            airbender_host::verify_real_proof_with_vk(proof, &vk, expected_output_commit)
103                .map_err(|err| CliError::with_source("proof verification failed", err))?;
104            proof.level()
105        }
106    };
107
108    if expected_output_words.is_none() {
109        tracing::warn!("public outputs were not provided; only proof/VK validity was checked");
110    }
111
112    ui::success("proof verified");
113    ui::field("level", host_level_name(level));
114    if let Some(words) = expected_output_words {
115        ui::field("expected_output", format_output_words(&words));
116    }
117
118    Ok(())
119}

121fn parse_expected_output_words(raw: Option<&str>) -> Result<Option<[u32; 8]>> {
122    let Some(raw) = raw else {
123        return Ok(None);
124    };
125
126    let trimmed = raw.trim();
127    if trimmed.is_empty() {
128        return Err(CliError::new("`--expected-output` cannot be empty")
129            .with_hint("provide comma-separated u32 words, for example `--expected-output 42`"));
130    }
131
132    let parts: Vec<&str> = trimmed.split(',').collect();
133    if parts.len() > 8 {
134        return Err(CliError::new(format!(
135            "`--expected-output` accepts at most 8 words (got {})",
136            parts.len()
137        ))
138        .with_hint(
139            "provide up to 8 comma-separated values for x10..x17; missing words are zero-padded",
140        ));
141    }
142
143    let mut words = [0u32; 8];
144    for (index, token) in parts.into_iter().enumerate() {
145        let token = token.trim();
146        if token.is_empty() {
147            return Err(CliError::new(format!(
148                "found an empty word at position {} in `--expected-output`",
149                index + 1
150            ))
151            .with_hint("use comma-separated values like `42,0,0`"));
152        }
153        words[index] = parse_output_word(token, index + 1)?;
154    }
155
156    Ok(Some(words))
157}

159fn parse_output_word(token: &str, position: usize) -> Result<u32> {
160    if let Some(hex) = token
161        .strip_prefix("0x")
162        .or_else(|| token.strip_prefix("0X"))
163    {
164        if hex.is_empty() {
165            return Err(CliError::new(format!(
166                "failed to parse output word at position {position}: `{token}`"
167            ))
168            .with_hint("hex output words must use `0x` followed by one or more hex digits"));
169        }
170
171        return u32::from_str_radix(hex, 16)
172            .map_err(|err| {
173                CliError::with_source(
174                    format!("failed to parse output word at position {position}: `{token}`"),
175                    err,
176                )
177            })
178            .map_err(|err| err.with_hint("use decimal or 0x-prefixed hexadecimal u32 words"));
179    }
180
181    token
182        .parse::<u32>()
183        .map_err(|err| {
184            CliError::with_source(
185                format!("failed to parse output word at position {position}: `{token}`"),
186                err,
187            )
188        })
189        .map_err(|err| err.with_hint("use decimal or 0x-prefixed hexadecimal u32 words"))
190}

192fn format_output_words(words: &[u32; 8]) -> String {
193    words
194        .iter()
195        .map(u32::to_string)
196        .collect::<Vec<_>>()
197        .join(",")
198}

200fn as_host_level(level: ProverLevelArg) -> airbender_host::ProverLevel {
201    match level {
202        ProverLevelArg::Base => airbender_host::ProverLevel::Base,
203        ProverLevelArg::RecursionUnrolled => airbender_host::ProverLevel::RecursionUnrolled,
204        ProverLevelArg::RecursionUnified => airbender_host::ProverLevel::RecursionUnified,
205    }
206}

212        ProverLevelArg::RecursionUnified => "recursion-unified",

224fn read_bincode<T: DeserializeOwned>(path: &Path) -> Result<T> {
225    let bytes = std::fs::read(path).map_err(|err| {
226        CliError::with_source(format!("failed to read `{}`", path.display()), err)
227    })?;
228    let (decoded, read_len): (T, usize) =
229        bincode::serde::decode_from_slice(&bytes, bincode::config::standard())
230            .map_err(|err| CliError::with_source("failed to decode bincode payload", err))?;
231
232    if read_len != bytes.len() {
233        tracing::warn!(
234            "bincode decoded {} bytes but file is {} bytes",
235            read_len,
236            bytes.len()
237        );
238    }
239    Ok(decoded)
240}
241
242fn write_bincode<T: Serialize>(path: &Path, value: &T) -> Result<()> {
243    let encoded = bincode::serde::encode_to_vec(value, bincode::config::standard())
244        .map_err(|err| CliError::with_source("failed to encode bincode payload", err))?;
245    std::fs::write(path, encoded).map_err(|err| {
246        CliError::with_source(format!("failed to write `{}`", path.display()), err)
247    })?;
248
249    Ok(())
250}
