9use crate::error::{CliError, Result};

11fn main() {
12    let cli = cli::Cli::parse_for_cargo();
13
14    if let Err(err) = init_tracing() {
15        ui::render_error(&err);
16        std::process::exit(1);
17    }
18
19    if let Err(err) = commands::run(cli) {
20        ui::render_error(&err);
21        std::process::exit(1);
22    }
23}
24
25fn init_tracing() -> Result<()> {
26    let filter = tracing_subscriber::EnvFilter::try_from_default_env()
27        .unwrap_or_else(|_| tracing_subscriber::EnvFilter::new("warn"));
28
29    tracing_subscriber::fmt()
30        .with_env_filter(filter)
31        .with_target(false)
32        .without_time()
33        .try_init()
34        .map_err(|err| {
35            CliError::with_source(
36                "failed to initialize tracing subscriber",
37                anyhow::Error::from_boxed(err),
38            )
39            .with_hint("set RUST_LOG=<level> (for example `RUST_LOG=info`) to inspect backend logs")
40        })?;
41
42    Ok(())
43}
