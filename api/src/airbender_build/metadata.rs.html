1//! Typed view of the guest project's `Cargo.toml` loaded from `cargo metadata`.

3use crate::errors::{BuildError, Result};
4use crate::Profile;
5use cargo_metadata::{Metadata, MetadataCommand, Package};
6use serde::Deserialize;
7use std::collections::HashMap;
8use std::path::Path;

10/// Combined cargo and airbender metadata for a guest project, loaded in a single
11/// `cargo metadata` invocation.
12pub(crate) struct CargoMetadata {
13    /// Cargo package name.
14    pub(crate) package_name: String,
15    /// Binary target names declared by the package.
16    pub(crate) bin_targets: Vec<String>,
17    /// Cargo workspace root reported by `cargo metadata`.
18    pub(crate) workspace_root: std::path::PathBuf,
19    /// Typed `[package.metadata.airbender]` settings, defaulting to empty if absent.
20    pub(crate) airbender: AirbenderConfig,
21}
22
23/// Contents of `[package.metadata.airbender]` in the guest `Cargo.toml`.
24#[derive(Deserialize, Default)]
25pub(crate) struct AirbenderConfig {
26    /// Per-profile build settings, keyed by profile name (`"debug"`, `"release"`).
27    #[serde(default)]
28    profile: HashMap<String, AirbenderProfileConfig>,
29}
30
31/// Per-profile airbender build settings under
32/// `[package.metadata.airbender.profile.<name>]`.
33#[derive(Deserialize, Default)]
34struct AirbenderProfileConfig {
35    /// Enable `panic_immediate_abort` build-std feature for this profile.
36    #[serde(default, rename = "panic-immediate-abort")]
37    panic_immediate_abort: bool,
38}

40fn load_metadata(manifest_path: &Path) -> Result<Metadata> {
41    load_metadata_with(manifest_path, None)
42}
43
44fn load_metadata_with(manifest_path: &Path, cargo_path: Option<&Path>) -> Result<Metadata> {
45    // `cargo metadata --manifest-path ...` still resolves rustup toolchain files from the
46    // subprocess cwd, so anchor the command to the guest project directory selected by
47    // `--project` instead of inheriting the caller's cwd.
48    let current_dir = manifest_path.parent().unwrap_or_else(|| Path::new("."));
49
50    let mut cmd = MetadataCommand::new();
51    cmd.manifest_path(manifest_path)
52        .current_dir(current_dir)
53        .no_deps();
54    if let Some(cargo_path) = cargo_path {
55        cmd.cargo_path(cargo_path);
56    }
57
58    cmd.exec()
59        .map_err(|err| BuildError::InvalidConfig(format!("cargo metadata failed: {err}")))
60}
61
62fn find_package<'a>(metadata: &'a Metadata, manifest_path: &Path) -> Result<&'a Package> {
63    let manifest_path = manifest_path.canonicalize()?;
64    let manifest_path =
65        cargo_metadata::camino::Utf8PathBuf::from_path_buf(manifest_path).map_err(|path| {
66            BuildError::InvalidConfig(format!(
67                "manifest path is not valid UTF-8: {}",
68                path.display()
69            ))
70        })?;
71
72    if let Some(pkg) = metadata
73        .packages
74        .iter()
75        .find(|pkg| pkg.manifest_path == manifest_path)
76    {
77        return Ok(pkg);
78    }
79
80    metadata
81        .root_package()
82        .ok_or(BuildError::MissingField("package.name"))
83}

85impl CargoMetadata {
86    /// Loads the guest project manifest from the `Cargo.toml` at `manifest_path`.
87    ///
88    /// Calls `cargo metadata` once and deserializes both cargo fields and
89    /// `[package.metadata.airbender]` settings. Unknown airbender keys are ignored.
90    pub(crate) fn load(manifest_path: &Path) -> Result<Self> {
91        let metadata = load_metadata(manifest_path)?;
92        let package = find_package(&metadata, manifest_path)?;
93        let bin_targets = package
94            .targets
95            .iter()
96            .filter(|t| t.kind.iter().any(|k| k == "bin"))
97            .map(|t| t.name.clone())
98            .collect();
99        let workspace_root = metadata.workspace_root.clone().into_std_path_buf();
100        let airbender =
101            serde_json::from_value(package.metadata["airbender"].clone()).unwrap_or_default();
102        Ok(Self {
103            package_name: package.name.clone(),
104            bin_targets,
105            workspace_root,
106            airbender,
107        })
108    }
109
110    /// Returns the effective `panic_immediate_abort` for the given profile.
111    ///
112    /// Reads `package.metadata.airbender.profile.<profile>.panic-immediate-abort`.
113    /// Defaults to `false` if the key is absent.
114    pub(crate) fn panic_immediate_abort(&self, profile: Profile) -> bool {
115        self.airbender
116            .profile
117            .get(profile.as_str())
118            .map(|p| p.panic_immediate_abort)
119            .unwrap_or(false)
120    }
121}

123#[cfg(test)]
124mod tests {
125    use super::*;
126    use std::fs;

129    fn make_metadata(json: &str) -> CargoMetadata {
130        CargoMetadata {
131            package_name: "guest".to_string(),
132            bin_targets: vec!["guest".to_string()],
133            workspace_root: std::path::PathBuf::new(),
134            airbender: serde_json::from_str(json).expect("parse airbender config"),
135        }
136    }
137
138    #[test]
139    fn panic_immediate_abort_reads_correct_profile() {
140        let m = make_metadata(r#"{"profile": {"release": {"panic-immediate-abort": true}}}"#);
141        assert!(m.panic_immediate_abort(Profile::Release));
142        assert!(!m.panic_immediate_abort(Profile::Debug));
143
144        let m = make_metadata(r#"{"profile": {"debug": {"panic-immediate-abort": true}}}"#);
145        assert!(!m.panic_immediate_abort(Profile::Release));
146        assert!(m.panic_immediate_abort(Profile::Debug));
147    }
148
149    #[test]
150    fn airbender_config_tolerates_missing_metadata() {
151        let null: AirbenderConfig =
152            serde_json::from_value(serde_json::Value::Null).unwrap_or_default();
153        assert!(null.profile.is_empty());
154        let unknown: AirbenderConfig = serde_json::from_str(r#"{"unknown-key": 42}"#).unwrap();
155        assert!(unknown.profile.is_empty());
156    }

158    #[test]
159    fn load_reads_package_name_and_bin_targets() {
160        let manifest_path = std::path::Path::new(env!("CARGO_MANIFEST_DIR"))
161            .parent()
162            .unwrap()
163            .parent()
164            .unwrap()
165            .join("examples/fibonacci/guest/Cargo.toml");
166        let m = CargoMetadata::load(&manifest_path).expect("load fibonacci manifest");
167        assert_eq!(m.package_name, "airbender-fibonacci");
168        assert_eq!(m.bin_targets, vec!["airbender-fibonacci"]);
169    }

171    #[test]
172    fn load_defaults_airbender_config_when_metadata_absent() {
173        let temp_dir = tempfile::tempdir().expect("create temp dir");
174        std::fs::write(
175            temp_dir.path().join("Cargo.toml"),

178        .expect("write Cargo.toml");
179        std::fs::create_dir(temp_dir.path().join("src")).expect("create src");
180        std::fs::write(temp_dir.path().join("src/main.rs"), "fn main() {}").expect("write main.rs");
181        let m = CargoMetadata::load(&temp_dir.path().join("Cargo.toml")).expect("load manifest");
182        assert!(!m.panic_immediate_abort(Profile::Release));
183        assert!(!m.panic_immediate_abort(Profile::Debug));

186    #[test]
187    fn load_runs_cargo_metadata_from_manifest_directory() {
188        let temp_dir = tempfile::tempdir().expect("create temp directory");
189        let project_dir = temp_dir.path().join("guest");
190        fs::create_dir_all(project_dir.join("src")).expect("create guest src");
191        fs::write(
192            project_dir.join("Cargo.toml"),
193            "[package]\nname = \"guest\"\nversion = \"0.1.0\"\nedition = \"2024\"\n",
194        )
195        .expect("write manifest");
196        fs::write(project_dir.join("src/main.rs"), "fn main() {}\n").expect("write main");
197
198        let log_path = temp_dir.path().join("cargo.cwd");
199        let cargo_wrapper = temp_dir.path().join("cargo-wrapper.sh");
200        let real_cargo = std::env::var_os("CARGO")
201            .map(std::path::PathBuf::from)
202            .unwrap_or_else(|| std::path::PathBuf::from("cargo"));
203        fs::write(
204            &cargo_wrapper,
205            format!(
206                "#!/bin/sh\nprintf '%s\\n' \"$PWD\" >> \"{}\"\nexec \"{}\" \"$@\"\n",
207                log_path.display(),
208                real_cargo.display()
209            ),
210        )
211        .expect("write cargo wrapper");
212        let mut permissions = fs::metadata(&cargo_wrapper)
213            .expect("read cargo wrapper metadata")
214            .permissions();
215        permissions.set_mode(0o755);
216        fs::set_permissions(&cargo_wrapper, permissions).expect("set cargo wrapper permissions");
217
218        load_metadata_with(&project_dir.join("Cargo.toml"), Some(&cargo_wrapper))
219            .expect("load metadata");
220
221        let cwd_log = fs::read_to_string(log_path).expect("read cargo cwd log");
222        assert_eq!(cwd_log.trim(), project_dir.display().to_string());
223    }
