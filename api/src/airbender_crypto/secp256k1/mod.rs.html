23pub enum Secp256k1Err {

29/// The order of the secp256k1 curve, divided by two. Signatures that should be checked according
30/// to EIP-2 should have an S value less than or equal to this.
31///
32/// `57896044618658097711785492504343953926418782139537452191302581570759080747168`
33pub const SECP256K1N_HALF: [u8; 32] = [
34    0x7F, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF,
35    0x5D, 0x57, 0x6E, 0x73, 0x57, 0xA4, 0x50, 0x1D, 0xDF, 0xE9, 0x2F, 0x46, 0x68, 0x1B, 0x20, 0xA0,
36];
37
38pub const SECP256K1N_HALF_U256: ruint::aliases::U256 =
39    ruint::aliases::U256::from_be_bytes(SECP256K1N_HALF);

41impl Display for Secp256k1Err {
42    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
43        match self {
44            Self::OperationOverflow => write!(f, "secp256k1: restoring x-coordinate overflowed"),
45            Self::InvalidParams => write!(
46                f,
47                "secp256k1: could not decompress signature to curve point"
48            ),
49            Self::RecoveredInfinity => write!(f, "secp256k1: recovered the point at infinity"),
50        }
51    }
52}

54#[cfg(feature = "secp256k1-static-context")]
55pub fn ecrecover_test() {
56    use crate::k256::{
57        ecdsa::{hazmat::bits2field, SigningKey},
58        elliptic_curve::{group::GroupEncoding, ops::Reduce},
59        Scalar,
60    };
61    use crate::sha3::Keccak256;
62    use crate::MiniDigest;
63    let message = "In the beginning the Universe was created.
64    This had made many people very angry and has been widely regarded as a bad move";
65    let private_key = SigningKey::from_bytes(
66        &[
67            136, 84, 181, 46, 13, 86, 203, 113, 63, 17, 137, 177, 95, 211, 104, 70, 112, 232, 200,
68            156, 225, 27, 123, 207, 243, 114, 4, 216, 148, 242, 81, 154,
69        ]
70        .into(),
71    )
72    .unwrap();
73    let digest = {
74        let mut hasher = Keccak256::new();
75        hasher.update(message);
76        let res = hasher.finalize();
77        let mut hash_bytes = [0u8; 32];
78        hash_bytes.copy_from_slice(&res);
79        hash_bytes
80    };
81
82    let public_key = private_key.verifying_key().as_affine();
83
84    let (signature, recovery_id) = private_key.sign_prehash_recoverable(&digest).unwrap();
85    let msg = <Scalar as Reduce<crate::k256::U256>>::reduce_bytes(
86        &bits2field::<crate::k256::Secp256k1>(&digest)
87            .map_err(|_| ())
88            .unwrap(),
89    );
90
91    let recovered_key = recover(&msg, &signature, &recovery_id).unwrap();
92
93    assert_eq!(recovered_key.to_bytes(), public_key.to_bytes());
94}
