41    pub(crate) const fn to_affine_const(self) -> AffineConst {
42        self.assert_verify();
43
44        if self.is_infinity() {
45            return AffineConst::INFINITY;
46        }
47
48        let zi = self.z.invert();
49        let z2 = zi.square();
50        let z3 = z2.mul(&zi);
51
52        AffineConst {
53            x: self.x.mul(&z2),
54            y: self.y.mul(&z3),
55            infinity: false,
56        }

63    // https://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-0.html#doubling-dbl-2009-l
64    pub(crate) const fn double(&self, rzr: Option<&mut FieldElementConst>) -> Self {
65        self.assert_verify();
66
67        if self.is_infinity() {
68            if let Some(rzr) = rzr {
69                *rzr = FieldElementConst::ONE;
70            }
71            return Self::INFINITY;
72        }
73
74        if let Some(rzr) = rzr {
75            *rzr = self.y.normalize().mul_int(2);
76        }
77
78        // A = X1^2
79        // B = Y1^2
80        // C = B^2
81        // D = 2*((X1+B)^2-A-C)
82        // E = 3*A
83        // F = E^2
84        // X3 = F-2*D
85        // Y3 = E*(D-X3)-8*C
86        // Z3 = 2*Y1*Z1
87
88        let a = self.x.square();
89        let b = self.y.square();
90        let c = b.square();
91        let d = self
92            .x
93            .add(&b)
94            .square()
95            .add(&a.negate(1))
96            .add(&c.negate(1))
97            .mul_int(2)
98            .normalize();
99        let e = a.mul_int(3);
100        let f = e.square();
101
102        let x = d.mul_int(2).normalize().negate(1).add(&f).normalize();
103
104        let g = x.negate(1).add(&d).mul(&e);
105        let y = g.add(&c.mul_int(8).normalize().negate(1));
106        let z = self.z.mul(&self.y).mul_int(2).normalize();
107
108        let ret = Self { x, y, z };
109
110        ret.assert_verify();
111
112        ret
113    }

115    // https://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-0.html#addition-madd-2007-bl
116    pub(crate) const fn add_ge(
117        &self,
118        a: &AffineConst,
119        rzr: Option<&mut FieldElementConst>,
120    ) -> Self {
121        self.assert_verify();
122        a.assert_verify();
123
124        if self.is_infinity() {
125            debug_assert!(rzr.is_none());
126            return a.to_jacobian();
127        }
128
129        if a.is_infinity() {
130            if let Some(rzr) = rzr {
131                *rzr = FieldElementConst::ONE
132            }
133
134            return *self;
135        }
136
137        let mut ret = JacobianConst::INFINITY;
138
139        let z12 = self.z.square();
140        let u1 = self.x;
141        let u2 = a.x.mul(&z12);
142        let s1 = self.y;
143        let s2 = a.y.mul(&z12).mul(&self.z);
144        let h = u1.negate(Self::X_MAGNITUDE_MAX).add(&u2);
145        let i = s2.negate(1).add(&s1);
146        if h.normalizes_to_zero() {
147            if i.normalizes_to_zero() {
148                ret = self.double(rzr);
149            } else if let Some(rzr) = rzr {
150                *rzr = FieldElementConst::ZERO;
151            }
152            return ret;
153        }
154
155        if let Some(rzr) = rzr {
156            *rzr = h;
157        }
158        ret.z = self.z.mul(&h);
159
160        let h2 = h.square().negate(1);
161        let mut h3 = h2.mul(&h);
162        let t = u1.mul(&h2);
163
164        ret.x = i.square().add(&h3).add(&t.mul_int(2)).normalize();
165
166        ret.y = t.add(&ret.x).mul(&i);
167        h3 = h3.mul(&s1);
168        ret.y = ret.y.add(&h3).normalize();
169
170        ret.assert_verify();
171
172        ret
173    }
174}

183impl Jacobian {
184    pub(crate) const INFINITY: Self = Self {
185        x: FieldElement::ZERO,
186        y: FieldElement::ONE,
187        z: FieldElement::ZERO,
188    };

190    const X_MAGNITUDE_MAX: u32 = 4;
191    const Y_MAGNITUDE_MAX: u32 = 4;
192    #[cfg(debug_assertions)]
193    const Z_MAGNITUDE_MAX: u32 = 1;

195    #[inline(always)]
196    pub(super) const fn assert_verify(&self) {
197        #[cfg(all(debug_assertions, not(feature = "bigint_ops")))]
198        {
199            debug_assert!(self.x.0.magnitude <= Self::X_MAGNITUDE_MAX);
200            debug_assert!(self.y.0.magnitude <= Self::Y_MAGNITUDE_MAX);
201            debug_assert!(self.z.0.magnitude <= Self::Z_MAGNITUDE_MAX);
202        }

205    pub(crate) fn is_infinity(&self) -> bool {
206        self.z.normalizes_to_zero()
207    }

209    pub(crate) fn to_affine(self) -> Affine {
210        self.assert_verify();
211
212        if self.is_infinity() {
213            return Affine::INFINITY;
214        }
215
216        let mut zi = self.z;
217        zi.invert_in_place();
218
219        let mut ret = Affine {
220            x: zi,
221            y: zi,
222            infinity: false,
223        };
224
225        ret.x.square_in_place();
226        ret.y *= ret.x;
227
228        ret.x *= self.x;
229        ret.y *= self.y;
230
231        ret
232    }

234    // this is essentially this algorithm https://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-0.html#doubling-dbl-2009-l
235    // but reorganized to reduce copies
236    pub(crate) fn double_in_place(&mut self, rzr: Option<&mut FieldElement>) {
237        self.assert_verify();
238
239        if self.is_infinity() {
240            if let Some(rzr) = rzr {
241                *rzr = FieldElement::ONE;
242            }
243
244            return;
245        }
246
247        if let Some(rzr) = rzr {
248            *rzr = self.y;
249            rzr.double_in_place();
250        }
251
252        self.z *= self.y;
253        self.z.double_in_place();
254        self.z.normalize_in_place();
255
256        let mut a = self.x;
257        a.square_in_place();
258
259        self.y.square_in_place();
260        let b = self.y;
261        self.y.square_in_place();
262        self.y.negate_in_place(1);
263
264        self.x += b;
265        self.x.square_in_place();
266        self.x -= a;
267        self.x += self.y;
268        self.x.double_in_place();
269        self.x.normalize_in_place();
270
271        let mut d = self.x;
272
273        a *= 3;
274        let mut f = a;
275        f.square_in_place();
276
277        self.x.double_in_place();
278        self.x.normalize_in_place();
279        self.x.negate_in_place(1);
280        self.x += f;
281        self.x.normalize_in_place();
282
283        d -= self.x;
284        d *= a;
285        self.y *= 8;
286        self.y.normalize_in_place();
287        self.y.add_in_place(&d);
288        self.y.normalize_in_place();
289
290        self.assert_verify();
291    }

293    // essentially this
294    // - https://github.com/bitcoin-core/secp256k1/blob/master/src/group_impl.h#L590
295    // - https://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-0.html#addition-madd
296    // Includes optimisations to reduce copies
297    pub(crate) fn add_ge_in_place(&mut self, mut a: Affine, rzr: Option<&mut FieldElement>) {
298        self.assert_verify();
299        a.assert_verify();
300
301        if self.is_infinity() {
302            debug_assert!(rzr.is_none());
303            *self = a.to_jacobian();
304            return;
305        }
306
307        if a.is_infinity() {
308            if let Some(rzr) = rzr {
309                *rzr = FieldElement::ONE;
310            }
311
312            return;
313        }
314
315        let mut z12 = self.z;
316        z12.square_in_place();
317        a.x *= z12;
318        a.y *= z12;
319        a.y *= self.z;
320        a.y.negate_in_place(1);
321
322        let mut h = self.x;
323        h.negate_in_place(Self::X_MAGNITUDE_MAX);
324        h += a.x;
325
326        let mut i = self.y;
327        i += a.y;
328
329        if h.normalizes_to_zero() {
330            if i.normalizes_to_zero() {
331                self.double_in_place(rzr);
332            } else {
333                if let Some(rzr) = rzr {
334                    *rzr = FieldElement::ZERO;
335                }
336                *self = Jacobian::INFINITY;
337            }
338            return;
339        }
340
341        if let Some(rzr) = rzr {
342            *rzr = h;
343        }
344
345        self.z *= h;
346        self.z.normalize_in_place();
347
348        let mut h2 = h;
349        h2.square_in_place();
350        h2.negate_in_place(1);
351        h *= h2;
352
353        self.x *= h2;
354        let mut t = self.x;
355        self.x.double_in_place();
356        self.x += h;
357        let mut i2 = i;
358        i2.square_in_place();
359        self.x += i2;
360        self.x.normalize_in_place();
361
362        t += self.x;
363        t *= i;
364        self.y *= h;
365        self.y += t;
366        self.y.normalize_in_place();
367    }

369    // https://github.com/bitcoin-core/secp256k1/blob/master/src/group_impl.h#L653
370    pub(crate) fn add_zinv_in_place(&mut self, mut b: Affine, z: &FieldElement) {
371        self.assert_verify();
372        b.assert_verify();
373
374        if b.is_infinity() {
375            return;
376        }
377
378        if self.is_infinity() {
379            let mut z2 = *z;
380            z2.square_in_place();
381            let mut z3 = z2;
382            z3 *= z;
383
384            self.x = b.x;
385            self.x *= z2;
386
387            self.y = b.y;
388            self.y *= z3;
389
390            self.z = FieldElement::ONE;
391
392            return;
393        }
394
395        //  We need to calculate (rx,ry,rz) = (ax,ay,az) + (bx,by,1/z). Due to
396        // secp256k1's isomorphism we can multiply the Z coordinates on both sides
397        // by z, and get: (rx,ry,rz*z) = (ax,ay,az*z) + (bx,by,1).
398        // This means that (rx,ry,rz) can be calculated as
399        // (ax,ay,az*z) + (bx,by,1), when not applying the z factor to rz.
400        // The variable az below holds the modified Z coordinate for a, which is used
401        // for the computation of rx and ry, but not for rz.
402        let mut az = self.z;
403        az.mul_in_place(z);
404
405        let mut z12 = az;
406        z12.square_in_place();
407
408        b.x *= z12;
409        b.y *= z12;
410        b.y *= az;
411
412        let mut h = self.x;
413        h.negate_in_place(Self::X_MAGNITUDE_MAX);
414        h += b.x;
415
416        let mut i = self.y;
417        i.negate_in_place(Self::Y_MAGNITUDE_MAX);
418        i += b.y;
419
420        if h.normalizes_to_zero() {
421            if i.normalizes_to_zero() {
422                self.double_in_place(None);
423            } else {
424                *self = Jacobian::INFINITY;
425            }
426            return;
427        }
428
429        self.z *= h;
430        self.z.normalize_in_place();
431
432        let mut h2 = h;
433        h2.square_in_place();
434        h *= h2;
435
436        let mut i2 = i;
437        i2.square_in_place();
438
439        self.x *= h2;
440        let t = self.x;
441        self.x.double_in_place();
442        self.x += h;
443        self.x.negate_in_place(Self::X_MAGNITUDE_MAX);
444        self.x += i2;
445        self.x.normalize_in_place();
446
447        self.y *= h;
448        self.y.negate_in_place(Self::Y_MAGNITUDE_MAX);
449        let mut x = self.x;
450        x.negate_in_place(Self::X_MAGNITUDE_MAX);
451        x += t;
452        x *= i;
453        self.y += x;
454        self.y.normalize_in_place();
455    }
